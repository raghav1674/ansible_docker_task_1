- hosts: slave
  vars_files:
  - vars.yml
  tasks:
  - name: configuring yum for docker
    yum_repository:
      name: docker
      description: Docker repo
      baseurl: https://download.docker.com/linux/centos/7/x86_64/stable
      gpgcheck: no

  - name: "installing Docker-CE"
    package:
      name:
      - "docker-ce-18.09.1-3.el7.x86_64"
      - python3
      state: present
    register: docker_ce

  - name: "Installing Docker SDK"
    pip:
      name: docker

  - name: "starting Docker service"
    service:
      name: docker
      state: started
    register: docker_svc

  - name: docker api
    lineinfile:
      path: /usr/lib/systemd/system/docker.service
      regexp: '^ExecStart'
      line: "ExecStart=/usr/bin/dockerd -H fd:// -H tcp://{{ ansible_enp0s8['ipv4']['address'] }}:4243"
    register: docker_api

  - debug:
      var: docker_api

  - name: restarting docker daemon after reload
    service:
      name: docker
      daemon_reload: yes
      state: restarted
      enabled: yes
    when: docker_api.changed == true


  - name: creating a base container having python installed
    docker_container:
      name: "{{ container_name }}"
      image: "{{ container_image }}"
      state: started
      ports:
      - "{{ port_to_expose }}:80"

  - name: creating an in memory ansible-host
    add_host:
      name: "{{ container_name }}"
      ansible_connection: docker
      ansible_docker_extra_args: "-H=tcp://{{ ansible_enp0s8['ipv4']['address'] }}:4243"
      groups:
      - container


- hosts: container
  vars_files:
  - vars.yml

  tasks:
  - name: installing httpd using package module
    package:
      name: httpd
      state: present
    register: z
    ignore_errors: yes

  - debug:
      var: z

  - name: installing httpd using command if package fails
    command: yum install httpd -y
    when: z.rc!=0


  - name: copying index file to root directory
    template:
      src: "{{ path_to_index_file }}"
      dest: "/var/www/html/ind.html"

    ignore_errors: yes

  - name: starting httpd service inside container
    command: "/usr/sbin/httpd"
